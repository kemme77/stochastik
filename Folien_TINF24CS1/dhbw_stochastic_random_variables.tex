\begin{frame}{\(\sigma\)-Algebra}
\begin{block}{Definition}
Eine Menge \(\mathcal A \subseteq \mathcal P(\Omega)\) hei\ss{}t \(\sigma\)-Algebra auf \(\Omega\), wenn gilt:
\begin{enumerate}
    \item \(\Omega \in \mathcal A\),
    \item aus \(A \in \mathcal A\) folgt \(A^{\mathsf c} \in \mathcal A\),
    \item aus \(A_1,A_2,\dots \in \mathcal A\) folgt
    \[
    \bigcup_{n\in\mathbb N} A_n \in \mathcal A.
    \]
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{Erzeugte \(\sigma\)-Algebra}
\begin{block}{Definition}
Für ein Mengensystem \(\mathcal G \subseteq \mathcal P(\Omega)\) bezeichnet
\[
\sigma(\mathcal G)
\]
die kleinste \(\sigma\)-Algebra auf \(\Omega\), die \(\mathcal G\) enthält.
\end{block}

\begin{block}{Idee}
Aus den Erzeugern in \(\mathcal G\) werden durch Komplementbildung und abzählbare Vereinigungen alle messbaren Mengen erzeugt.
\end{block}
\end{frame}

\begin{frame}{Existenz und Eindeutigkeit der erzeugten \(\sigma\)-Algebra}
\begin{block}{Satz}
Sei \(\mathcal G\subseteq \mathcal P(\Omega)\) ein Mengensystem.
Dann gibt es genau eine kleinste \(\sigma\)-Algebra auf \(\Omega\), die \(\mathcal G\) enthält.
\end{block}

\begin{block}{Beweisidee}
Betrachte
\[
\mathfrak F
=
\{\mathcal A\subseteq\mathcal P(\Omega):
\mathcal A \text{ ist \(\sigma\)-Algebra und }
\mathcal G\subseteq\mathcal A\}.
\]
Diese Familie ist nicht leer, denn \(\mathcal P(\Omega)\in\mathfrak F\).
\end{block}
\end{frame}

\begin{frame}{Existenz und Eindeutigkeit (Fortsetzung)}
\begin{block}{Beweisidee (Forts.)}
Setze
\[
\sigma(\mathcal G):=\bigcap_{\mathcal A\in\mathfrak F}\mathcal A.
\]
Ein Durchschnitt von \(\sigma\)-Algebren ist wieder eine \(\sigma\)-Algebra und enthält \(\mathcal G\).
Außerdem gilt für jede \(\sigma\)-Algebra \(\mathcal B\) mit \(\mathcal G\subseteq\mathcal B\):
\[
\sigma(\mathcal G)\subseteq\mathcal B.
\]
Also ist \(\sigma(\mathcal G)\) die kleinste solche \(\sigma\)-Algebra.
Die Eindeutigkeit folgt aus der Minimalität.
\end{block}
\end{frame}

\begin{frame}{Topologischer Raum}
\begin{block}{Definition}
Ein topologischer Raum ist ein Paar \((X,\tau)\), wobei \(X\) eine Menge ist und
\(\tau\subseteq\mathcal P(X)\) die folgenden Eigenschaften erfüllt:
\begin{enumerate}
    \item \(\varnothing\in\tau\) und \(X\in\tau\),
    \item für beliebig viele \(U_i\in\tau\) gilt \(\bigcup_{i\in I} U_i\in\tau\),
    \item für endlich viele \(U_1,\dots,U_n\in\tau\) gilt \(U_1\cap\cdots\cap U_n\in\tau\).
\end{enumerate}
Die Elemente von \(\tau\) hei\ss{}en offene Mengen.
\end{block}
\end{frame}

\begin{frame}{Standardtopologie auf \(\mathbb R^n\)}
\begin{block}{Definition}
Für \(x\in\mathbb R^n\) und \(\varepsilon>0\) ist die offene Kugel
\[
B_\varepsilon(x):=\{y\in\mathbb R^n : \|y-x\|<\varepsilon\}.
\]
Eine Menge \(U\subseteq\mathbb R^n\) hei\ss{}t offen, wenn es zu jedem \(x\in U\) ein \(\varepsilon>0\) gibt mit \(B_\varepsilon(x)\subseteq U\).
\end{block}

\begin{block}{Standardtopologie}
Die Menge aller offenen Mengen in \(\mathbb R^n\) bildet eine Topologie, die Standardtopologie auf \(\mathbb R^n\).
\end{block}
\end{frame}

\begin{frame}{Borelsche \(\sigma\)-Algebra auf topologischen Räumen}
\begin{block}{Allgemeine Definition}
Sei \(X\) ein topologischer Raum mit Topologie \(\tau\).
Dann hei\ss{}t
\[
\mathcal B(X):=\sigma(\tau)
\]
die borelsche \(\sigma\)-Algebra auf \(X\).
\end{block}

\begin{block}{Bedeutung}
Die borelsche \(\sigma\)-Algebra ist die kleinste \(\sigma\)-Algebra, die alle offenen Mengen enthält.
\end{block}
\end{frame}

\begin{frame}{Borelsche \(\sigma\)-Algebra: Beispiele}
\begin{block}{Beispiele}
Für \(X=\mathbb R\) mit der üblichen Topologie erhält man
\[
\mathcal B(\mathbb R)=\sigma(\mathcal O),
\]
wobei \(\mathcal O\) die Menge der offenen Mengen in \(\mathbb R\) ist.
Allgemein ebenso
\[
\mathcal B(\mathbb R^n)=\sigma(\text{offene Mengen in }\mathbb R^n).
\]
\end{block}
\end{frame}

\begin{frame}{Borelsche \(\sigma\)-Algebra auf \(\mathbb R\)}
\begin{block}{Spezialfall}
Für den topologischen Raum \(\mathbb R\) mit der üblichen Topologie ist
\[
\mathcal B(\mathbb R)=\sigma(\mathcal O),
\]
wobei \(\mathcal O\) die offenen Mengen in \(\mathbb R\) bezeichnet.
\end{block}

\begin{block}{Wichtige Erzeuger}
Äquivalent gilt
\[
\mathcal B(\mathbb R)=\sigma\bigl(\{(-\infty,a] : a\in\mathbb R\}\bigr).
\]
\end{block}
\end{frame}

\begin{frame}{Messbarkeit reellwertiger Funktionen}
\begin{block}{Definition}
Seien \((\Omega,\mathcal A)\) und \((E,\mathcal E)\) Messräume .
Eine Abbildung
\[
X :\Omega \to E
\]
hei\ss{}t messbar, wenn für alle \(e \in \mathcal E \) gilt:
\[
X^{-1}(e)\in\mathcal A.
\]
\end{block}
\end{frame}

\begin{frame}{Messbarkeit über Erzeuger prüfen}
\begin{block}{Satz}
Seien \((\Omega,\mathcal A)\) und \((E,\mathcal E)\) Messräume und sei
\[
\mathcal E=\sigma(\mathcal G)
\]
für ein Mengensystem \(\mathcal G\subseteq\mathcal P(E)\).
Dann gilt für eine Abbildung
\[
X\colon \Omega\to E:
\]
\(X\) ist genau dann \(\mathcal A/\mathcal E\)-messbar, wenn für alle \(G\in\mathcal G\) gilt:
\[
X^{-1}(G)\in\mathcal A.
\]
\end{block}

\begin{block}{Interpretation}
Wenn die Ziel-\(\sigma\)-Algebra von einem Erzeugersystem erzeugt wird, dann reicht es, die Urbilder der Erzeuger zu prüfen.
\end{block}
\end{frame}

\begin{frame}{Beweis: Messbarkeit über Erzeuger (1/3)}
\begin{block}{Hinrichtung}
Die Richtung
\[
X \text{ messbar } \Rightarrow X^{-1}(G)\in\mathcal A
\quad\text{für alle }G\in\mathcal G
\]
ist klar, denn \(\mathcal G\subseteq\mathcal E\).
\end{block}

\begin{block}{Rückrichtung: Ansatz}
Für die Umkehrung setze
\[
\mathcal D:=\{B\subseteq E : X^{-1}(B)\in\mathcal A\}.
\]
Behauptung: \(\mathcal D\) ist eine \(\sigma\)-Algebra auf \(E\).
\end{block}
\end{frame}

\begin{frame}{Beweis: Messbarkeit über Erzeuger (2/3)}
\begin{block}{\(\mathcal D\) ist eine \(\sigma\)-Algebra}
\begin{enumerate}
    \item \(X^{-1}(E)=\Omega\in\mathcal A\), also \(E\in\mathcal D\),
    \item aus \(B\in\mathcal D\) folgt
    \[
    X^{-1}(B^{\mathsf c})=(X^{-1}(B))^{\mathsf c}\in\mathcal A,
    \]
    also \(B^{\mathsf c}\in\mathcal D\),
    \item für \(B_1,B_2,\dots\in\mathcal D\) gilt
    \[
    X^{-1}\!\left(\bigcup_{n\in\mathbb N} B_n\right)
    =
    \bigcup_{n\in\mathbb N} X^{-1}(B_n)\in\mathcal A,
    \]
    also \(\bigcup_{n\in\mathbb N} B_n\in\mathcal D\).
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{Beweis: Messbarkeit über Erzeuger (3/3)}
\begin{block}{Schlussfolgerung}
Nach Voraussetzung gilt \(X^{-1}(G)\in\mathcal A\) für alle \(G\in\mathcal G\), also
\[
\mathcal G\subseteq\mathcal D.
\]
Da \(\mathcal D\) eine \(\sigma\)-Algebra ist, die \(\mathcal G\) enthält, folgt
\[
\sigma(\mathcal G)\subseteq\mathcal D.
\]
Wegen \(\mathcal E=\sigma(\mathcal G)\) gilt also \(\mathcal E\subseteq\mathcal D\).

Damit ist für jedes \(B\in\mathcal E\): \(X^{-1}(B)\in\mathcal A\).
Also ist \(X\) messbar.
\end{block}
\end{frame}

\begin{frame}{Messbarkeit reellwertiger Funktionen}
\begin{block}{Definition}
Eine Abbildung
\[
X\colon \Omega \to \mathbb R
\]
hei\ss{}t messbar, wenn für alle \(B\in\mathcal B(\mathbb R)\) gilt:
\[
X^{-1}(B)\in\mathcal A.
\]
\end{block}

\begin{block}{Praktisches Kriterium}
Oft genügt es zu zeigen:
\[
\{\omega\in\Omega : X(\omega)\le a\}\in\mathcal A
\qquad
\text{für alle } a\in\mathbb R.
\]
\end{block}
\end{frame}

\begin{frame}{Maß}
\begin{block}{Definition}
Ein Maß auf einem Messraum \((\Omega,\mathcal A)\) ist eine Abbildung
\[
\mu\colon \mathcal A \to [0,\infty]
\]
mit:
\begin{enumerate}
    \item \(\mu(A)\ge 0\) für alle \(A\in\mathcal A\),
    \item \(\mu(\varnothing)=0\),
    \item für paarweise disjunkte \(A_1,A_2,\dots \in \mathcal A\) gilt
    \[
    \mu\!\left(\bigcup_{n\in\mathbb N} A_n\right)
    =
    \sum_{n\in\mathbb N}\mu(A_n).
    \]
\end{enumerate}
\end{block}
\end{frame}


\begin{frame}{Stetigkeit von Maßen}
\begin{block}{Satz}
Sei \(\mu\) ein Maß auf \((\Omega,\mathcal A)\) und seien \(A_1\subseteq A_2\subseteq\cdots\) Mengen in \(\mathcal A\) mit
\[
A_n\uparrow A:=\bigcup_{n\in\mathbb N} A_n.
\]
Dann gilt
\[
\mu(A_n)\to\mu(A).
\]
\end{block}

\begin{block}{Anschaulich}
Wächst eine Folge von Mengen, so wächst auch deren Maß gegen das Maß der Vereinigung.
\end{block}
\end{frame}

\begin{frame}{Beweis: Stetigkeit von Maßen}
\begin{block}{Idee: Teleskopsumme}
Definiere \(B_1:=A_1\) und \(B_n:=A_n\setminus A_{n-1}\) für \(n\ge 2\).
\\[4pt]
Die \(B_n\) sind paarweise disjunkt und es gilt
\[
A_n=B_1\cup\cdots\cup B_n
\qquad\text{und}\qquad
A=\bigcup_{n=1}^\infty B_n.
\]
\end{block}

\begin{block}{Rechnung}
Mit der \(\sigma\)-Additivität folgt
\[
\mu(A)=\sum_{n=1}^\infty\mu(B_n)
=\lim_{N\to\infty}\sum_{n=1}^N\mu(B_n)
=\lim_{N\to\infty}\mu(A_N).
\]
\end{block}
\end{frame}

\begin{frame}{Restringiertes Maß}
\begin{block}{Definition}
Sei \(\mu\) ein Maß auf \((\Omega,\mathcal A)\) und \(s\in\mathcal A\).
Das auf \(s\) restringierte Maß \(\mu\vert_s\) ist definiert durch
\[
\mu\vert_s(A):=\mu(A\cap s),
\qquad A\in\mathcal A.
\]
\end{block}


\begin{center}
   \includegraphics[scale=0.07]{img/bedingtewkeit.jpeg} 
\end{center}

\end{frame}

\begin{frame}{Restringiertes Maß: Eigenschaften}
\begin{block}{Maßeigenschaft}
\(\mu\vert_s\) ist wieder ein Maß auf \((\Omega,\mathcal A)\):
\begin{itemize}
    \item \(\mu\vert_s(\varnothing)=\mu(\varnothing\cap s)=0\).
    \item Für paarweise disjunkte \((A_n)\):
    \begin{align*}
    \mu\vert_s\!\left(\bigcup_n A_n\right)
    &=\mu\!\left(\bigcup_n (A_n\cap s)\right)\\
    &=\sum_n \mu(A_n\cap s)
    =\sum_n \mu\vert_s(A_n).
    \end{align*}
\end{itemize}
\end{block}

\begin{block}{Monotonie}
Falls \(s\subseteq t\), dann \(\mu\vert_s(A)\le\mu\vert_t(A)\) für alle \(A\in\mathcal A\).
\\[4pt]
{\small\texttt{Measure.restrict\_mono (hs : s \(\subseteq\) t) :}}\\
{\small\texttt{\quad \(\mu\).restrict s \(\le\) \(\mu\).restrict t}}
\end{block}
\end{frame}

\begin{frame}{Wahrscheinlichkeitsmaß}
\begin{block}{Definition}
Ein Wahrscheinlichkeitsmaß ist ein Maß \(\mathbb P\) auf \((\Omega,\mathcal A)\) mit
\[
\mathbb P(\Omega)=1.
\]
\end{block}

\begin{block}{Interpretation}
Ein Maß misst allgemein die Größe messbarer Mengen.
Ein Wahrscheinlichkeitsmaß ist der Spezialfall, in dem die Gesamtmasse des ganzen Raums gleich \(1\) ist.
\end{block}
\end{frame}


\begin{frame}{Beispiel: diskretes Wahrscheinlichkeitsmaß}
\begin{block}{Definition}
Für eine endliche Menge \(\Omega=\{\omega_1,\ldots,\omega_n\}\) und Wahrscheinlichkeiten
\(p_1,\ldots,p_n>0\) mit \(\sum_{i=1}^n p_i=1\) definiert man
\[
\mathbb P(A)=\sum_{\omega\in A} p_\omega.
\]
\end{block}

\begin{block}{Spezialfall: Gleichverteilung}
Falls \(p_i=\frac1n\) für alle \(i\), dann gilt
\[
\mathbb P(A)=\frac{|A|}{n}.
\]
\end{block}
\end{frame}

\begin{frame}{Bedingte Wahrscheinlichkeit}
\begin{block}{Definition}
Sei \((\Omega,\mathcal A,\mathbb P)\) ein Wahrscheinlichkeitsraum und \(B\in\mathcal A\) mit \(\mathbb P(B)>0\).
Die bedingte Wahrscheinlichkeit von \(A\) gegeben \(B\) ist
\[
\mathbb P(A\mid B):=\frac{\mathbb P(A\cap B)}{\mathbb P(B)}.
\]
\end{block}

\begin{block}{Interpretation}
\(\mathbb P(A\mid B)\) ist die Wahrscheinlichkeit von \(A\), wenn man bereits weiß, dass \(B\) eingetreten ist.
\end{block}
\end{frame}

\begin{frame}{Bedingte Wahrscheinlichkeit als Maß}
\begin{block}{Satz}
Für festes \(B\in\mathcal A\) mit \(\mathbb P(B)>0\) ist die Abbildung
\[
\mathbb P(\cdot\mid B)\colon \mathcal A\to[0,1],
\quad
A\mapsto \mathbb P(A\mid B)
\]
ein Wahrscheinlichkeitsmaß auf \((\Omega,\mathcal A)\).
\end{block}

\begin{block}{Zusammenhang mit restringiertem Maß}
Es gilt
\[
\mathbb P(A\mid B)=\frac{\mathbb P\vert_B(A)}{\mathbb P(B)}.
\]
Die bedingte Wahrscheinlichkeit ist das normierte restringierte Maß.
\end{block}
\end{frame}


\begin{frame}{Zufallsvariable}
\begin{block}{Definition}
Seien \((\Omega,\mathcal A,\mathbb P)\) ein Wahrscheinlichkeitsraum und \((E,\mathcal E)\) ein Messraum.
Eine Abbildung
\[
X\colon \Omega \to E
\]
hei\ss{}t Zufallsvariable, wenn
\[
X^{-1}(B)\in\mathcal A
\qquad
\text{für alle } B\in\mathcal E.
\]
\end{block}
\end{frame}

\begin{frame}{Induzierte Verteilung}
\begin{block}{Definition}
Ist
\[
X\colon (\Omega,\mathcal A,\mathbb P)\to(E,\mathcal E)
\]
eine Zufallsvariable, so ist die induzierte Verteilung von \(X\) das Maß \(\mathbb P_X\) auf \((E,\mathcal E)\) mit
\[
\mathbb P_X(B)=\mathbb P(X\in B)=\mathbb P\bigl(X^{-1}(B)\bigr),
\qquad B\in\mathcal E.
\]
\end{block}
\end{frame}

\begin{frame}{Bildmaß}
\begin{block}{Definition}
Seien \((X,\mathcal A,\mu)\) und \((Y,\mathcal B)\) Messräume und
\[
T\colon X\to Y
\]
eine messbare Abbildung.
Dann definiert man das Bildmaß \(T_\#\mu\) auf \((Y,\mathcal B)\) durch
\[
T_\#\mu(B):=\mu(T^{-1}(B)),
\qquad B\in\mathcal B.
\]
\end{block}

\begin{block}{Spezialfall}
Ist
\[
X\colon (\Omega,\mathcal F,\mathbb P)\to(E,\mathcal E)
\]
eine Zufallsvariable, dann ist ihre Verteilung gerade das Bildmaß
\[
\mathbb P_X=X_\#\mathbb P.
\]
\end{block}
\end{frame}

\begin{frame}{Integration: Grundidee}
\begin{block}{Ziel}
Zu einer messbaren Funktion
\[
f\colon \Omega\to\mathbb R
\]
und einem Maß \(\mu\) möchte man eine Zahl
\[
\int_\Omega f\,d\mu
\]
definieren.
\end{block}

\begin{block}{Lebesgue-Idee}
Nicht das Definitionsgebiet wird zerlegt, sondern die Funktion wird durch einfache Stufenfunktionen approximiert.
\end{block}
\end{frame}

\begin{frame}{Indikatorfunktion}
\begin{block}{Definition}
Für eine Menge \(A\subseteq\Omega\) ist die Indikatorfunktion definiert durch
\[
\mathbf 1_A(\omega):=
\begin{cases}
1, & \omega\in A,\\
0, & \omega\notin A.
\end{cases}
\]
\end{block}

\begin{block}{Messbarkeit}
Ist \(A\in\mathcal A\), dann ist \(\mathbf 1_A\) messbar.
\end{block}
\end{frame}

\begin{frame}{Einfache Funktion}
\begin{block}{Definition}
Eine messbare Funktion
\[
\varphi\colon \Omega\to\mathbb R
\]
hei\ss{}t einfach, wenn sie nur endlich viele Werte annimmt.
\end{block}

\begin{block}{Darstellung}
Dann gilt
\[
\varphi=\sum_{i=1}^n a_i\,\mathbf 1_{A_i},
\qquad A_i\in\mathcal A.
\]
\end{block}

\begin{block}{Bemerkung}
Nichtnegative einfache Funktionen sind der Spezialfall \(a_i\ge 0\).
\end{block}
\end{frame}

\begin{frame}{Nichtnegative einfache Funktionen}
\begin{block}{Definition}
Eine nichtnegative einfache Funktion hat die Form
\[
\varphi=\sum_{i=1}^n a_i\,\mathbf 1_{A_i},
\qquad
a_i\ge 0,\ A_i\in\mathcal A.
\]
\end{block}

\begin{block}{Integral}
Für eine solche Funktion setzt man
\[
\int_\Omega \varphi\,d\mu
=
\sum_{i=1}^n a_i\,\mu(A_i).
\]
\end{block}

\begin{block}{Bezug zu Lean}
In mathlib entspricht das dem Begriff der \emph{simple function}.
\end{block}
\end{frame}

\begin{frame}{Wohldefiniertheit für einfache Funktionen}
\begin{block}{Problem}
Eine einfache Funktion kann verschiedene Darstellungen haben:
\[
\varphi=\sum_{i=1}^n a_i\,\mathbf 1_{A_i}
\qquad\text{und}\qquad
\varphi=\sum_{j=1}^m b_j\,\mathbf 1_{B_j}.
\]
\end{block}

\begin{block}{Behauptung}
Der Wert
\[
\int_\Omega \varphi\,d\mu
=
\sum_{i=1}^n a_i\,\mu(A_i)
\]
hängt nicht von der Darstellung ab.
\end{block}
\end{frame}

\begin{frame}{Wohldefiniertheit: Beweisidee}
\begin{block}{Idee}
Man zerlegt den Raum in die Schnitte
\[
C_{ij}=A_i\cap B_j.
\]
Auf jedem \(C_{ij}\) haben beide Darstellungen denselben konstanten Wert, also liefern sie dieselbe Summe.
\end{block}
\end{frame}

\begin{frame}{Nichtnegative messbare Funktionen}
\begin{block}{Definition}
Sei
\[
f\colon \Omega\to[0,\infty]
\]
messbar.
Dann definiert man
\[
\int_\Omega f\,d\mu
:=
\sup\left\{
\int_\Omega \varphi\,d\mu :
0\le \varphi\le f,\ \varphi \text{ einfach}
\right\}.
\]
\end{block}

\begin{block}{Warum ist das eindeutig?}
Die Menge aller solchen Approximationen ist durch \(f\) und \(\mu\) eindeutig bestimmt. Also ist auch ihr Supremum eindeutig bestimmt.
\end{block}
\end{frame}

\begin{frame}{Monotonie des Integrals}
\begin{block}{Satz}
Seien \(f,g\colon\Omega\to[0,\infty]\) messbar mit \(f\le g\). Dann gilt
\[
\int_\Omega f\,d\mu\le\int_\Omega g\,d\mu.
\]
\end{block}

\begin{block}{Beweis}
Jede einfache Funktion \(\varphi\) mit \(0\le\varphi\le f\) erfüllt auch \(0\le\varphi\le g\). Also ist
\[
\left\{\int\varphi\,d\mu : 0\le\varphi\le f\right\}
\subseteq
\left\{\int\varphi\,d\mu : 0\le\varphi\le g\right\}.
\]
Das Supremum über eine Teilmenge ist \(\le\) dem Supremum über die Obermenge.
\end{block}
\end{frame}

\begin{frame}{Integration über eine Menge}
\begin{block}{Definition}
Für \(A\in\mathcal A\) und eine messbare Funktion \(f\colon\Omega\to\mathbb R\) setzt man
\[
\int_A f\,d\mu
:=
\int_\Omega f\cdot\mathbf 1_A\,d\mu.
\]
\end{block}

\begin{block}{Interpretation}
Man integriert nicht über den ganzen Raum, sondern nur über die Menge \(A\), indem man \(f\) außerhalb von \(A\) durch Null ersetzt.
\end{block}
\end{frame}

\begin{frame}{Approximation durch Treppenfunktionen}
\begin{block}{Satz}
Sei
\[
f\colon \Omega\to[0,\infty]
\]
messbar. Dann gibt es eine Folge nichtnegativer einfacher Funktionen
\[
\varphi_n\colon\Omega\to[0,\infty)
\]
mit
\[
0\le \varphi_n\le \varphi_{n+1}\le f
\qquad\text{und}\qquad
\varphi_n(\omega)\uparrow f(\omega)
\]
für alle \(\omega\in\Omega\).
\end{block}

\begin{block}{Interpretation}
Jede nichtnegative messbare Funktion kann von unten durch Treppenfunktionen approximiert werden.
\end{block}
\end{frame}

\begin{frame}{Konstruktion der Approximation}
\begin{block}{Definition der Treppenfunktionen}
Für \(n\in\mathbb N\) setze
\[
\varphi_n(\omega)
=
\sum_{k=0}^{n2^n-1}
\frac{k}{2^n}\,
\mathbf 1_{\left\{
\frac{k}{2^n}\le f(\omega)<\frac{k+1}{2^n}
\right\}}
+
n\,\mathbf 1_{\{f(\omega)\ge n\}}.
\]
\end{block}

\begin{block}{Idee}
Man rundet \(f(\omega)\) nach unten auf Stufen der Breite \(2^{-n}\) und schneidet bei Höhe \(n\) ab.
\end{block}
\end{frame}

\begin{frame}{Warum ist \(\varphi_n\le\varphi_{n+1}\)?}
\begin{block}{Idee}
Beim Übergang von \(n\) zu \(n+1\) passieren zwei Dinge:
\begin{enumerate}
    \item Die Stufenbreite halbiert sich: \(2^{-n}\to 2^{-(n+1)}\).
    \item Die Abschneidehöhe steigt: \(n\to n+1\).
\end{enumerate}
\end{block}

\begin{block}{Fall 1: \(f(\omega)<n\)}
Dann wird \(f(\omega)\) weder bei Stufe \(n\) noch bei \(n+1\) abgeschnitten.
\\[4pt]
\(\varphi_n(\omega)\) rundet \(f(\omega)\) auf ein Vielfaches von \(2^{-n}\) ab.
\\
\(\varphi_{n+1}(\omega)\) rundet auf ein Vielfaches von \(2^{-(n+1)}\) ab.
\\[4pt]
Jedes Vielfache von \(2^{-n}\) ist auch ein Vielfaches von \(2^{-(n+1)}\), denn \(\tfrac{k}{2^n}=\tfrac{2k}{2^{n+1}}\).
Das feinere Raster hat aber zusätzliche Zwischenwerte. Also kann die feinere Abrundung nur gleich oder größer sein.
\end{block}
\end{frame}

\begin{frame}{Warum ist \(\varphi_n\le\varphi_{n+1}\)? -- Beispiel}
\begin{block}{Konkretes Beispiel}
Sei \(f(\omega)=2{,}7\).
\begin{itemize}
    \item \(n=1\): Stufenbreite \(2^{-1}=0{,}5\). Abrundung: \(\varphi_1(\omega)=2{,}5\).
    \item \(n=2\): Stufenbreite \(2^{-2}=0{,}25\). Abrundung: \(\varphi_2(\omega)=2{,}5\).
    \item \(n=3\): Stufenbreite \(2^{-3}=0{,}125\). Abrundung: \(\varphi_3(\omega)=2{,}625\).
\end{itemize}
\end{block}

\begin{block}{Beobachtung}
Bei \(n=2\) bleibt der Wert gleich, weil kein neuer Rasterpunkt dazukommt.
\\
Bei \(n=3\) rückt ein neuer Rasterpunkt (\(2{,}625\)) näher an \(2{,}7\) heran.
\\[4pt]
Allgemein: Die Abrundung kann beim Verfeinern nur steigen oder gleich bleiben -- nie fallen.
\end{block}
\end{frame}

\begin{frame}{Warum ist \(\varphi_n\le\varphi_{n+1}\)? (Forts.)}
\begin{block}{Fall 2: \(n\le f(\omega)<n+1\)}
Hier wird bei Stufe \(n\) abgeschnitten: \(\varphi_n(\omega)=n\).
\\
Bei Stufe \(n+1\) wird \(f(\omega)\) auf Vielfache von \(2^{-(n+1)}\) abgerundet, also \(\varphi_{n+1}(\omega)\ge n\).
\end{block}

\begin{block}{Fall 3: \(f(\omega)\ge n+1\)}
Dann \(\varphi_n(\omega)=n\) und \(\varphi_{n+1}(\omega)=n+1\).
\end{block}

\begin{block}{Fazit}
In allen Fällen gilt \(\varphi_n(\omega)\le \varphi_{n+1}(\omega)\).
\end{block}
\end{frame}

\begin{frame}[shrink=5]{Beweis der Approximation}
\begin{block}{Messbarkeit und Einfachheit}
Die Mengen
\[
\left\{
\frac{k}{2^n}\le f<\frac{k+1}{2^n}
\right\}
\qquad\text{und}\qquad
\{f\ge n\}
\]
sind messbar, da \(f\) messbar ist.

Außerdem nimmt \(\varphi_n\) nur endlich viele Werte an. Also ist \(\varphi_n\) eine einfache Funktion.
\end{block}

\begin{block}{Abschätzung}
Nach Konstruktion gilt für alle \(\omega\in\Omega\):
\[
0\le \varphi_n(\omega)\le f(\omega).
\]
Außerdem werden die Stufen feiner und die Abschneidehöhe größer, also
\[
\varphi_n(\omega)\le \varphi_{n+1}(\omega).
\]
\end{block}
\end{frame}

\begin{frame}{Punktweise Konvergenz}
\begin{block}{Fall \(f(\omega)<\infty\)}
Sei \(f(\omega)<\infty\). Für \(n>f(\omega)\) wird nicht mehr abgeschnitten.
Dann ist \(\varphi_n(\omega)\) die Abrundung von \(f(\omega)\) auf Stufen der Breite \(2^{-n}\).

Daher gilt
\[
0\le f(\omega)-\varphi_n(\omega)<2^{-n}.
\]
Also folgt
\[
\varphi_n(\omega)\to f(\omega).
\]
\end{block}

\begin{block}{Fall \(f(\omega)=\infty\)}
Ist \(f(\omega)=\infty\), dann gilt
\[
\varphi_n(\omega)=n,
\]
also
\[
\varphi_n(\omega)\to\infty=f(\omega).
\]
\end{block}
\end{frame}

\begin{frame}{Reellwertige messbare Funktionen}
\begin{block}{Folgerung}
Ist
\[
f\colon\Omega\to\mathbb R
\]
messbar, dann gibt es einfache Funktionen \(s_n\) mit
\[
s_n(\omega)\to f(\omega)
\]
für alle \(\omega\in\Omega\).
\end{block}
\end{frame}

\begin{frame}{Reellwertige messbare Funktionen (Forts.)}
\begin{block}{Begründung}
Zerlege
\[
f=f^+-f^-.
\]
Für \(f^+\) und \(f^-\) gibt es nichtnegative einfache Funktionen
\[
\varphi_n\uparrow f^+,
\qquad
\psi_n\uparrow f^-.
\]
Dann ist
\[
s_n:=\varphi_n-\psi_n
\]
einfach und es gilt
\[
s_n\to f^+-f^-=f.
\]
\end{block}
\end{frame}

\begin{frame}{Monotone Approximation des Integrals}
\begin{block}{Satz}
Sei
\[
f\colon \Omega\to[0,\infty]
\]
messbar und \((\varphi_n)_{n\in\mathbb N}\) eine Folge nichtnegativer messbarer Funktionen mit
\[
\varphi_n(\omega)\le \varphi_{n+1}(\omega)
\qquad\text{für alle } \omega\in\Omega
\]
und
\[
\varphi_n(\omega)\uparrow f(\omega)
\qquad\text{für alle } \omega\in\Omega.
\]
Dann gilt
\[
\int_\Omega f\,d\mu
=
\lim_{n\to\infty}\int_\Omega \varphi_n\,d\mu.
\]
\end{block}
\end{frame}

\begin{frame}[shrink=5]{Warum darf man so das Integral berechnen?}
\begin{block}{Idee}
Wächst eine Folge \((\varphi_n)\) punktweise von unten gegen \(f\), dann wachsen auch die Integrale
\[
\int_\Omega \varphi_n\,d\mu
\]
monoton an.
\end{block}

\begin{block}{Satz von der monotonen Konvergenz}
Der Grenzwert dieser Integrale ist genau das Integral der Grenzfunktion:
\[
\lim_{n\to\infty}\int_\Omega \varphi_n\,d\mu
=
\int_\Omega f\,d\mu.
\]
\end{block}

\begin{block}{Nutzen}
Damit kann man \(\int f\,d\mu\) in der Praxis durch eine geeignete Folge
\(\varphi_n\uparrow f\) berechnen.
\end{block}
\end{frame}

\begin{frame}{Satz von der monotonen Konvergenz}
\begin{block}{Satz}
Seien \(f_n\colon\Omega\to[0,\infty]\) messbar mit
\[
f_n\uparrow f.
\]
Dann gilt
\[
\int_\Omega f\,d\mu
=
\lim_{n\to\infty}\int_\Omega f_n\,d\mu.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisübersicht}
\begin{block}{Hauptargumente}
\begin{enumerate}
\item \textbf{``\(\le\)'':} Aus \(f_n\le f\) folgt sofort \(\lim\int f_n\,d\mu\le\int f\,d\mu\) (Monotonie des Integrals).
\item \textbf{``\(\ge\)'':} Für jede einfache Funktion \(0\le\varphi\le f\) und \(0<c<1\) definiere \(A_n=\{f_n\ge c\varphi\}\). Dann:
\begin{itemize}
\item \(A_n\uparrow\Omega\), da \(f_n\uparrow f > c\varphi\).
\item Auf \(A_n\): \ \(\int f_n\,d\mu\ge c\int_{A_n}\varphi\,d\mu\).
\item Da \(\varphi\) einfach: \(\int_{A_n}\varphi\,d\mu\to\int_\Omega\varphi\,d\mu\) (Stetigkeit des Maßes, endliche Summe).
\end{itemize}
\item \textbf{Abschluss:} \(c\uparrow 1\) und Supremum über alle \(\varphi\le f\) liefert \(\lim\int f_n\,d\mu\ge\int f\,d\mu\).
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{Beweisidee: eine Richtung}
\begin{block}{Monotonie}
Da \(f_n\le f\), gilt
\[
\int_\Omega f_n\,d\mu\le \int_\Omega f\,d\mu.
\]
Also
\[
\lim_{n\to\infty}\int_\Omega f_n\,d\mu
\le
\int_\Omega f\,d\mu.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisidee: andere Richtung}
\begin{block}{Approximation von unten}
Sei \(\varphi\) eine einfache Funktion mit
\[
0\le \varphi\le f.
\]
Für \(0<c<1\) betrachte
\[
A_n:=\{\,\omega: f_n(\omega)\ge c\varphi(\omega)\,\}.
\]
Da \(f_n(\omega)\uparrow f(\omega)\) und \(c\varphi(\omega)<\varphi(\omega)\le f(\omega)\), gilt
\[
A_n\uparrow\Omega.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisidee: Integralabschätzung}
\begin{block}{Auf \(A_n\)}
Auf \(A_n\) gilt
\[
f_n\ge c\varphi.
\]
Daher
\[
\int_\Omega f_n\,d\mu
\ge
\int_{A_n} f_n\,d\mu
\ge
c\int_{A_n}\varphi\,d\mu.
\]
\end{block}

\begin{block}{Grenzübergang}
Da \(\varphi\) einfach ist und \(A_n\uparrow\Omega\), gilt
\[
\int_{A_n}\varphi\,d\mu
\to
\int_\Omega \varphi\,d\mu.
\]
Also
\[
\lim_{n\to\infty}\int_\Omega f_n\,d\mu
\ge
c\int_\Omega \varphi\,d\mu.
\]
\end{block}
\end{frame}

\begin{frame}{Warum reicht hier die Einfachheit von \(\varphi\)?}
\begin{block}{Problem}
Im Beweis brauchen wir \(\int_{A_n}\varphi\,d\mu\to\int_\Omega\varphi\,d\mu\) für \(A_n\uparrow\Omega\).
\\[4pt]
Dafür dürfen wir nicht den Satz der monotonen Konvergenz verwenden -- den beweisen wir ja gerade!
\end{block}

\begin{block}{Lösung: endliche Summe}
Da \(\varphi\) einfach ist, gilt \(\varphi=\sum_{i=1}^m a_i\,\mathbf 1_{B_i}\). Also:
\[
\int_{A_n}\varphi\,d\mu
=\sum_{i=1}^m a_i\,\mu(A_n\cap B_i).
\]
\end{block}
\end{frame}

\begin{frame}{Warum reicht hier die Einfachheit? (Forts.)}
\begin{block}{Grenzübergang}
Wegen \(A_n\uparrow\Omega\) gilt \(A_n\cap B_i\uparrow B_i\), und die Stetigkeit des Maßes liefert
\[
\mu(A_n\cap B_i)\to\mu(B_i).
\]
Da die Summe \emph{endlich} ist, darf man den Grenzwert gliedweise bilden:
\[
\sum_{i=1}^m a_i\,\mu(A_n\cap B_i)\to\sum_{i=1}^m a_i\,\mu(B_i)=\int_\Omega\varphi\,d\mu.
\]
\end{block}

\begin{block}{Fazit}
Die Einfachheit von \(\varphi\) vermeidet einen Zirkelschluss: Man braucht nur die Stetigkeit des Maßes und den Grenzwert einer endlichen Summe -- nicht die monotone Konvergenz selbst.
\end{block}
\end{frame}

\begin{frame}{Beweisabschluss}
\begin{block}{Grenze \(c\uparrow 1\)}
Für jedes \(0<c<1\) gilt
\[
\lim_{n\to\infty}\int_\Omega f_n\,d\mu
\ge
c\int_\Omega \varphi\,d\mu.
\]
Da die linke Seite nicht von \(c\) abhängt, können wir das Supremum über \(0<c<1\) bilden:
\[
\lim_{n\to\infty}\int_\Omega f_n\,d\mu
\ge
\sup_{0<c<1} c\int_\Omega \varphi\,d\mu
=
\int_\Omega \varphi\,d\mu.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisabschluss (Forts.)}
\begin{block}{Supremum über einfache Funktionen}
Da dies für jede einfache Funktion \(0\le\varphi\le f\) gilt,
\[
\lim_{n\to\infty}\int_\Omega f_n\,d\mu
\ge
\sup_{\substack{0\le\varphi\le f\\ \varphi\ \text{einfach}}}
\int_\Omega \varphi\,d\mu
=
\int_\Omega f\,d\mu.
\]
Zusammen mit der anderen Ungleichung folgt die Behauptung.
\end{block}
\end{frame}


\begin{frame}{Eindeutigkeit des Werts bei Approximationen}
\begin{block}{Satz}
Seien \((\varphi_n)\) und \((\psi_n)\) zwei Folgen nichtnegativer messbarer Funktionen mit
\[
\varphi_n\uparrow f
\qquad\text{und}\qquad
\psi_n\uparrow f.
\]
Dann gilt
\[
\lim_{n\to\infty}\int_\Omega \varphi_n\,d\mu
=
\lim_{n\to\infty}\int_\Omega \psi_n\,d\mu.
\]
\end{block}

\begin{block}{Begründung}
Beide Grenzwerte sind nach dem Satz von der monotonen Konvergenz gleich
\[
\int_\Omega f\,d\mu.
\]
\end{block}
\end{frame}

\begin{frame}{Reellwertige Funktionen}
\begin{block}{Zerlegung}
Für eine reellwertige Funktion \(f\) setzt man
\[
f^+(\omega)=\max\{f(\omega),0\},
\qquad
f^-(\omega)=\max\{-f(\omega),0\}.
\]
Dann gilt
\[
f=f^+-f^-,
\qquad
|f|=f^++f^-.
\]
\end{block}

\begin{block}{Definition}
Man setzt
\[
\int_\Omega f\,d\mu
=
\int_\Omega f^+\,d\mu-\int_\Omega f^-\,d\mu,
\]
sofern nicht beide Terme gleichzeitig unendlich sind.
\end{block}
\end{frame}

\begin{frame}{Integrierbarkeit}
\begin{block}{Definition}
Eine reellwertige messbare Funktion \(f\) hei\ss{}t integrierbar, wenn
\[
\int_\Omega |f|\,d\mu<\infty.
\]
\end{block}

\begin{block}{Folge}
Dann sind \(\int f^+\,d\mu\) und \(\int f^-\,d\mu\) endlich, und damit ist \(\int f\,d\mu\) wohldefiniert und eindeutig.
\end{block}

\begin{block}{Bezug zu Lean}
In mathlib schreibt man dafür \texttt{Integrable f $\mu$}.
\end{block}
\end{frame}


\begin{frame}{Notation in Lean/mathlib}
\begin{block}{Nichtnegatives Integral}
Für Funktionen
\[
f\colon \Omega\to[0,\infty]
\]
verwendet mathlib die Notation
\[
\int^- \omega, f(\omega)\,\partial\mu.
\]
\end{block}

\begin{block}{Gewöhnliches Integral}
Für reellwertige Funktionen schreibt man
\[
\int \omega, f(\omega)\,\partial\mu.
\]
\end{block}
\end{frame}

\begin{frame}{Diskreter Spezialfall}
\begin{block}{Endlicher Fall}
Ist \(\Omega\) endlich und \(\mu\) ein diskretes Maß, dann gilt
\[
\int_\Omega f\,d\mu
=
\sum_{\omega\in\Omega} f(\omega)\,\mu(\{\omega\}).
\]
\end{block}

\begin{block}{Für Wahrscheinlichkeitsmaße}
Ist \(\mu=\mathbb P\) und $X$ eine reelle Zufallsvariable, dann ist
\[
\mathbb E[X]=\int_\Omega X\,d\mathbb P
\]
der Erwartungswert von $X$. Außerdem gilt für jede messbare Menge \(A\):
\[
\mathbb E[\mathbf{1}_A]=\int_\Omega \mathbf{1}_A\,d\mathbb P = \mathbb P(A).
\]
\end{block}
\end{frame}

\begin{frame}{Erwartungswert im diskreten Fall}
\begin{block}{Formel}
Ist \(X\colon\Omega\to\mathbb R\) eine Zufallsvariable auf einem diskreten Wahrscheinlichkeitsraum
\((\Omega,\mathcal A,\mathbb P)\) mit abzählbarem \(\Omega\), so gilt
\[
\mathbb E[X]=\sum_{\omega\in\Omega} X(\omega)\,\mathbb P(\{\omega\}).
\]
\end{block}

\begin{block}{Bildraum-Formel}
Hat \(X\) die Werte \(x_1,x_2,\ldots\) mit \(\mathbb P(X=x_i)=p_i\), so gilt
\[
\mathbb E[X]=\sum_{i} x_i\,p_i = \sum_{i} x_i\,\mathbb P(X=x_i).
\]
\end{block}
\end{frame}

\begin{frame}{Messbare Rechtecke}
\begin{block}{Definition}
Für Teilmengen \(A\subseteq E\) und \(B\subseteq F\) hei\ss{}t
\[
A\times B:=\{(x,y)\in E\times F : x\in A,\ y\in B\}
\]
ein Rechteck im Produktraum \(E\times F\).
\end{block}

\begin{block}{Beispiel}
Falls \(E=F=\mathbb R\) und \(A,B\) Intervalle sind, dann ist \(A\times B\subseteq\mathbb R^2\)
ein gewöhnliches Rechteck.
\end{block}
\end{frame}

\begin{frame}{Produkt-\(\sigma\)-Algebra}
\begin{block}{Definition}
Für Messräume \((E,\mathcal E)\) und \((F,\mathcal F)\) ist die Produkt-\(\sigma\)-Algebra
\[
\mathcal E\otimes\mathcal F
=
\sigma\bigl(\{A\times B : A\in\mathcal E,\; B\in\mathcal F\}\bigr).
\]
\end{block}

\begin{block}{Idee}
Sie ist die natürliche \(\sigma\)-Algebra auf dem Produktraum \(E\times F\).
\end{block}
\end{frame}

\begin{frame}{Gemeinsame Verteilung}
\begin{block}{Definition}
Sind
\[
X\colon \Omega\to E,
\qquad
Y\colon \Omega\to F
\]
Zufallsvariablen, so ist
\[
(X,Y)\colon \Omega\to E\times F,
\qquad
\omega\mapsto (X(\omega),Y(\omega)),
\]
eine Zufallsvariable auf \((E\times F,\mathcal E\otimes\mathcal F)\).
\end{block}

\begin{block}{Gemeinsame Verteilung}
Die induzierte Verteilung von \((X,Y)\) hei\ss{}t gemeinsame Verteilung:
\[
\mathbb P_{(X,Y)}(C)=\mathbb P\bigl((X,Y)\in C\bigr),
\qquad
C\in\mathcal E\otimes\mathcal F.
\]
\end{block}
\end{frame}

\begin{frame}{Produktmaß}
\begin{block}{Satz}
Seien \(\mu\) ein Maß auf \((E,\mathcal E)\) und \(\nu\) ein Maß auf \((F,\mathcal F)\).
Dann gibt es genau ein Maß
\[
\mu\otimes\nu
\]
auf \((E\times F,\mathcal E\otimes\mathcal F)\) mit
\[
(\mu\otimes\nu)(A\times B)=\mu(A)\nu(B)
\]
für alle \(A\in\mathcal E\), \(B\in\mathcal F\).
\end{block}
\end{frame}

\begin{frame}{Produktmaß über Schnitte}
\begin{block}{Schnittfunktion}
Für eine messbare Menge \(S\subseteq E\times F\) und \(x\in E\) setze
\[
S_x:=\{y\in F:(x,y)\in S\}.
\]
\end{block}

\begin{block}{Idee}
Dann kann man das Produktmaß durch
\[
(\mu\otimes\nu)(S)
=
\int_E \nu(S_x)\,d\mu(x)
\]
beschreiben.
\end{block}
\end{frame}

\begin{frame}{Produktmaß über Schnitte (Forts.)}
\begin{block}{Interpretation}
Man misst für jedes \(x\) den vertikalen Schnitt und integriert diese Größen über \(x\).
\end{block}
\end{frame}

\begin{frame}{Produktmaß auf Rechtecken}
\begin{block}{Spezialfall}
Für \(S=A\times B\) gilt
\[
S_x=
\begin{cases}
B,& x\in A,\\
\varnothing,& x\notin A.
\end{cases}
\]
Also
\[
\nu(S_x)=
\begin{cases}
\nu(B),& x\in A,\\
0,& x\notin A.
\end{cases}
\]
\end{block}

\begin{block}{Folgerung}
Daher
\[
(\mu\otimes\nu)(A\times B)
=
\int_E \nu(S_x)\,d\mu(x)
=
\nu(B)\mu(A).
\]
\end{block}
\end{frame}

\begin{frame}{Lebesgue-Maß auf \(\mathbb R\)}
\begin{block}{Satz/Definition}
Es gibt genau ein Maß \(\lambda\) auf \((\mathbb R,\mathcal B(\mathbb R))\) mit
\[
\lambda((a,b])=b-a
\qquad\text{für alle } a<b.
\]
Dieses Maß hei\ss{}t Lebesgue-Maß auf \(\mathbb R\).
\end{block}

\begin{block}{Interpretation}
\(\lambda\) misst die Länge von Intervallen und allgemeiner von borelschen Mengen.
\end{block}
\end{frame}

\begin{frame}{Beispiele für das Lebesgue-Maß auf \(\mathbb R\)}
\begin{block}{Intervalle}
Für Intervalle gilt insbesondere
\[
\lambda([a,b])=b-a,
\qquad
\lambda((a,b))=b-a.
\]
\end{block}

\begin{block}{Einpunktmengen}
Außerdem gilt
\[
\lambda(\{x\})=0
\qquad\text{für alle } x\in\mathbb R.
\]
\end{block}
\end{frame}

\begin{frame}{Motivation in Dimension 1}
\begin{block}{Riemann-Idee für nette Mengen}
Für Intervalle und andere reguläre Mengen kann man Länge anschaulich durch
\[
\lambda(A)=\int_{\mathbb R} \mathbf 1_A(x)\,dx
\]
als Riemann-Integral der Indikatorfunktion motivieren.
\end{block}

\begin{block}{Grenze dieser Idee}
Das funktioniert nicht für alle messbaren Mengen, denn \(\mathbf 1_A\) ist im Allgemeinen
nicht Riemann-integrierbar.
\newline
Beispiel: Für \(A=\mathbb Q\cap[0,1]\) ist \(\mathbf 1_A\) nicht Riemann-integrierbar,
obwohl \(\lambda(A)=0\) gilt.
\end{block}
\end{frame}

\begin{frame}{Lebesgue-Maß auf \(\mathbb R^n\)}
\begin{block}{Definition}
Hat man das Lebesgue-Maß \(\lambda\) auf \(\mathbb R\) bereits konstruiert, so definiert man das Lebesgue-Maß auf \(\mathbb R^n\) als \(n\)-faches Produktmaß
\[
\lambda^n=\lambda\otimes\cdots\otimes\lambda.
\]
\end{block}

\begin{block}{Idee}
Das \(1\)-dimensionale Lebesgue-Maß misst Längen.
Das Produktmaß liefert daraus in höheren Dimensionen Flächen und Volumina.
\end{block}
\end{frame}

\begin{frame}{Volumenmaß auf Quadern}
\begin{block}{Spezialfall}
Für einen Quader
\[
[a_1,b_1]\times\cdots\times[a_n,b_n]\subseteq\mathbb R^n
\]
gilt
\[
\lambda^n\bigl([a_1,b_1]\times\cdots\times[a_n,b_n]\bigr)
=
\prod_{i=1}^n (b_i-a_i).
\]
\end{block}

\begin{block}{Interpretation}
In Dimension \(1\) ist das Länge, in Dimension \(2\) Fläche, in Dimension \(3\) Volumen.
\end{block}
\end{frame}

\begin{frame}{Satz von Fubini}
\begin{block}{Satz}
Seien \((E,\mathcal E,\mu)\) und \((F,\mathcal F,\nu)\) Maßräume und
\[
f\colon E\times F\to\mathbb R
\]
messbar und integrierbar. Dann gilt:
\[
\int_{E\times F} f(x,y)\,d(\mu\otimes\nu)(x,y)
=
\int_E\left(\int_F f(x,y)\,d\nu(y)\right)d\mu(x).
\]
Ebenso gilt
\[
\int_{E\times F} f(x,y)\,d(\mu\otimes\nu)(x,y)
=
\int_F\left(\int_E f(x,y)\,d\mu(x)\right)d\nu(y).
\]
\end{block}
\end{frame}

\begin{frame}{Beweisidee zu Fubini}
\begin{block}{Schritt 1: Indikatorfunktionen}
Für \(f=\mathbf 1_S\) mit messbarem \(S\subseteq E\times F\) gilt
\[
\int_{E\times F}\mathbf 1_S\,d(\mu\otimes\nu)
=
(\mu\otimes\nu)(S),
\]
und nach der Definition des Produktmaßes
\[
(\mu\otimes\nu)(S)=\int_E \nu(S_x)\,d\mu(x).
\]
Da \(\nu(S_x)=\int_F \mathbf 1_S(x,y)\,d\nu(y)\), folgt die Behauptung.
\end{block}

\begin{block}{Schritt 2: Einfache Funktionen}
Mit Linearität folgt der Satz für endliche Summen von Indikatorfunktionen, also für einfache Funktionen.
\end{block}
\end{frame}

\begin{frame}{Beweisidee zu Fubini: Grenzübergang}
\begin{block}{Schritt 3: Nichtnegative messbare Funktionen}
Man approximiert \(f\ge 0\) von unten durch einfache Funktionen.
Dann erhält man die Aussage mit dem Satz von der monotonen Konvergenz.
\end{block}

\begin{block}{Schritt 4: Allgemeine integrierbare Funktionen}
Für integrierbare \(f\) schreibt man
\[
f=f^+-f^-.
\]
Die Aussage für \(f^+\) und \(f^-\) ist bereits bekannt.
Durch Subtraktion folgt der Satz für \(f\).
\end{block}

\begin{block}{Bemerkung}
Für nichtnegative Funktionen spricht man oft vom Satz von Tonelli, für integrierbare Funktionen vom Satz von Fubini.
\end{block}
\end{frame}


\begin{frame}{Dichte eines Maßes}
\begin{block}{Definition}
Sei \(\mu\) ein Maß auf \((\mathbb R^n,\mathcal B(\mathbb R^n))\).
Eine messbare Funktion
\[
p\colon \mathbb R^n\to[0,\infty]
\]
heißt Dichte von \(\mu\) bezüglich des Lebesgue-Maßes \(\lambda^n\), wenn für alle messbaren Mengen \(A\subseteq\mathbb R^n\) gilt
\[
\mu(A)=\int_A p(x)\,d\lambda^n(x).
\]
\end{block}
\end{frame}

\begin{frame}{Schreibweise für Dichten}
\begin{block}{Kurzschreibweise}
Hat \(\mu\) die Dichte \(p\) bezüglich \(\lambda^n\), so schreibt man auch
\[
d\mu(x)=p(x)\,d\lambda^n(x).
\]
In \(\mathbb R^n\) schreibt man häufig kurz
\[
d\mu(x)=p(x)\,dx.
\]
\end{block}

\begin{block}{Bedeutung von \(dx\)}
Das Symbol \(dx\) steht hier also für Integration bezüglich des Lebesgue-Maßes.
\end{block}
\end{frame}
\begin{frame}{Beispiel: Gleichverteilung auf einem Intervall}
\begin{block}{Beispiel}
Für \([a,b]\subseteq\mathbb R\) mit \(a<b\) ist die Gleichverteilung auf \([a,b]\)
das Wahrscheinlichkeitsmaß mit Dichte
\[
p(x)=\frac1{b-a}\mathbf 1_{[a,b]}(x).
\]
\end{block}


\end{frame}

\begin{frame}{Transformationssatz für allgemeine Maße}
\begin{block}{Satz}
Seien \((X,\mathcal A,\mu)\), \((Y,\mathcal B)\) Messräume,
\[
T\colon X\to Y
\]
messbar und \(T_\#\mu\) das zugehörige Bildmaß.
Dann gilt für jede messbare Funktion
\[
f\colon Y\to\mathbb R
\]
mit \(f\ge 0\) oder \(f\in L^1(T_\#\mu)\):
\[
\int_Y f(y)\,d(T_\#\mu)(y)
=
\int_X f(T(x))\,d\mu(x).
\]
\end{block}
\begin{block}{Erinnerung}
\[
(T_\#\mu)(B)=\mu(T^{-1}(B))
\qquad\text{für alle } B\in\mathcal B.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zum Transformationssatz I}
\begin{block}{Schritt 1: Indikatorfunktionen}
Sei \(f=\mathbf 1_B\) mit \(B\in\mathcal B\). Dann ist
\[
\int_Y \mathbf 1_B(y)\,d(T_\#\mu)(y)=T_\#\mu(B).
\]
Nach Definition des Bildmaßes gilt
\[
T_\#\mu(B)=\mu(T^{-1}(B)).
\]
Andererseits ist
\[
\mathbf 1_B(T(x))=
\mathbf 1_{T^{-1}(B)}(x),
\]
also
\[
\int_X \mathbf 1_B(T(x))\,d\mu(x)
=
\int_X \mathbf 1_{T^{-1}(B)}(x)\,d\mu(x)
=
\mu(T^{-1}(B)).
\]
Damit ist die Behauptung für Indikatorfunktionen gezeigt.
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zum Transformationssatz II}
\begin{block}{Schritt 2: Einfache Funktionen}
Sei nun
\[
f=\sum_{i=1}^n a_i\,\mathbf 1_{B_i}
\]
eine nichtnegative einfache Funktion auf \(Y\).
Dann folgt mit Linearität:
\[
\int_Y f\,d(T_\#\mu)
=
\sum_{i=1}^n a_i \int_Y \mathbf 1_{B_i}\,d(T_\#\mu).
\]
Für jeden Summanden gilt bereits der Indikatorfall, also
\[
\int_Y \mathbf 1_{B_i}\,d(T_\#\mu)
=
\int_X \mathbf 1_{B_i}(T(x))\,d\mu(x).
\]
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zum Transformationssatz II (Forts.)}
\begin{block}{Schritt 2 (Forts.)}
Somit
\[
\int_Y f\,d(T_\#\mu)
=
\int_X \sum_{i=1}^n a_i\,\mathbf 1_{B_i}(T(x))\,d\mu(x)
=
\int_X f(T(x))\,d\mu(x).
\]
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zum Transformationssatz III}
\begin{block}{Schritt 3: Nichtnegative messbare Funktionen}
Sei \(f\colon Y\to[0,\infty]\) messbar.
Wähle eine wachsende Folge einfacher Funktionen \(f_n\) mit
\[
f_n\uparrow f.
\]
Dann gilt auch
\[
f_n\circ T \uparrow f\circ T.
\]
Mit dem Satz von der monotonen Konvergenz folgt
\begin{align*}
\int_Y f\,d(T_\#\mu)
&=
\lim_{n\to\infty}\int_Y f_n\,d(T_\#\mu)\\
&=
\lim_{n\to\infty}\int_X f_n(T(x))\,d\mu(x)\\
&=
\int_X f(T(x))\,d\mu(x).
\end{align*}
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zum Transformationssatz IV}
\begin{block}{Schritt 4: Allgemeine integrierbare Funktionen}
Ist \(f\in L^1(T_\#\mu)\), so zerlegt man
\[
f=f^+-f^-.
\]
Die Aussage gilt bereits für die nichtnegativen Funktionen \(f^+\) und \(f^-\). Also
\[
\int_Y f^+\,d(T_\#\mu)
=
\int_X f^+(T(x))\,d\mu(x),
\]
\[
\int_Y f^-\,d(T_\#\mu)
=
\int_X f^-(T(x))\,d\mu(x).
\]
Durch Subtraktion erhält man
\[
\int_Y f\,d(T_\#\mu)
=
\int_X f(T(x))\,d\mu(x).
\]
\end{block}
\end{frame}

\begin{frame}{Dichtetransformation in \(\mathbb R^n\)}
\begin{block}{Satz}
Sei \(T\colon U\to V\) eine \(C^1\)-Diffeomorphie zwischen offenen Mengen
\(U,V\subseteq\mathbb R^n\).
Sei \(\mu\) ein Maß auf \(U\) mit Dichte \(p\), also
\[
d\mu(x)=p(x)\,d\lambda^n(x).
\]
Dann besitzt das Bildmaß \(T_\#\mu\) auf \(V\) die Dichte
\[
q(y)=p(T^{-1}(y))
\frac{1}{|\det DT(T^{-1}(y))|}.
\]
\end{block}
\end{frame}

\begin{frame}{Transformationsformel in \(\mathbb R^n\)}
\begin{block}{Folgerung}
Für jede messbare Funktion \(f\colon V\to\mathbb R\) mit \(f\ge 0\) oder \(f\in L^1(T_\#\mu)\) gilt
\[
\int_V f(y)\,q(y)\,d\lambda^n(y)
=
\int_U f(T(x))\,p(x)\,d\lambda^n(x),
\]
wobei
\[
q(y)=p(T^{-1}(y))
\frac{1}{|\det DT(T^{-1}(y))|}.
\]
\end{block}

\begin{block}{Kurzform}
Mit \(d\lambda^n(x)=dx\) und \(d\lambda^n(y)=dy\) schreibt man auch
\[
\int_V f(y)\,q(y)\,dy
=
\int_U f(T(x))\,p(x)\,dx.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zur Dichtetransformation I}
\begin{block}{Ausgangspunkt}
Nach dem allgemeinen Transformationssatz gilt für messbares \(f\):
\[
\int_V f(y)\,d(T_\#\mu)(y)
=
\int_U f(T(x))\,d\mu(x).
\]
Da \(\mu\) die Dichte \(p\) besitzt, folgt
\[
\int_V f(y)\,d(T_\#\mu)(y)
=
\int_U f(T(x))\,p(x)\,d\lambda^n(x).
\]
\end{block}

\begin{block}{Ziel}
Wir möchten die linke Seite wieder als Integral gegen eine Dichte \(q\) auf \(V\) schreiben.
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zur Dichtetransformation II}
\begin{block}{Lokale lineare Approximation}
In einem kleinen Gebiet um \(x\) wird \(T\) näherungsweise durch die lineare Abbildung
\[
DT(x)
\]
beschrieben.
Eine lineare Abbildung \(A\) skaliert Volumina mit dem Faktor
\[
|\det A|.
\]
Daher wird ein kleines Volumenelement \(d\lambda^n(x)\) unter \(T\) näherungsweise zu
\[
d\lambda^n(y)\approx |\det DT(x)|\,d\lambda^n(x).
\]
\end{block}
\end{frame}

\begin{frame}{Beweisskizze zur Dichtetransformation III}
\begin{block}{Einsetzen von \(x=T^{-1}(y)\)}
Setzt man \(x=T^{-1}(y)\), dann erhält man heuristisch
\[
p(x)\,d\lambda^n(x)
=
p(T^{-1}(y))
\frac{1}{|\det DT(T^{-1}(y))|}\,d\lambda^n(y).
\]
Daher ist die Dichte des Bildmaßes
\[
q(y)=p(T^{-1}(y))
\frac{1}{|\det DT(T^{-1}(y))|}.
\]
\end{block}

\begin{block}{Bemerkung}
Ein strenger Beweis benötigt den mehrdimensionalen Transformationssatz aus der Analysis.
\end{block}
\end{frame}

\begin{frame}{Substitutionssatz in \(\mathbb R^n\)}
\begin{block}{Herleitung des Jacobi-Faktors}
Setze \(S:=T^{-1}\colon V\to U\). Auf \(V\) ist \(\mu=\lambda^n\), also in der
allgemeinen Dichtetransformation \(p\equiv 1\). Daher hat \(S_\#\lambda^n\) auf \(U\)
die Dichte
\[
q(x)=\frac{1}{|\det DS(S^{-1}(x))|}.
\]
Da \(S^{-1}=T\) und \(DS=D(T^{-1})\), ist
\[
q(x)=\frac{1}{|\det D(T^{-1})(T(x))|}.
\]
Wegen
\[
\left|\det D(T^{-1})(T(x))\right|=\frac{1}{|\det DT(x)|}
\]
folgt
\[
q(x)=|\det DT(x)|.
\]
\end{block}
\end{frame}

\begin{frame}{Substitutionssatz in \(\mathbb R^n\) (Forts.)}
\begin{block}{Spezialfall Lebesgue-Maß}
Also ist \(S_\#\lambda^n\) das Maß mit Dichte \(|\det DT(x)|\) bezüglich \(\lambda^n\).
Für messbares \(f\) gilt daher
\[
\int_V f(y)\,d\lambda^n(y)
=
\int_U f(T(x))\,|\det DT(x)|\,d\lambda^n(x).
\]
\end{block}

\begin{block}{Kurzform}
Mit \(d\lambda^n(x)=dx\) und \(d\lambda^n(y)=dy\) schreibt man
\[
\int_V f(y)\,dy
=
\int_U f(T(x))\,|\det DT(x)|\,dx.
\]
\end{block}
\end{frame}

\begin{frame}{Spezialfall in einer Dimension}
\begin{block}{Satz}
Ist \(T\colon [a,b]\to[T(a),T(b)]\) streng monoton und \(C^1\), dann gilt
\[
\int_{T(a)}^{T(b)} f(y)\,dy
=
\int_a^b f(T(x))\,|T'(x)|\,dx.
\]
\end{block}

\begin{block}{Monoton wachsender Fall}
Ist \(T\) streng wachsend, dann
\[
\int_{T(a)}^{T(b)} f(y)\,dy
=
\int_a^b f(T(x))\,T'(x)\,dx.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisskizze in Dimension 1}
\begin{block}{Schritt 1: Intervalle}
Zunächst für \(f=\mathbf 1_{(u,v]}\).
Wegen der Monotonie ist
\[
T^{-1}((u,v])
\]
wieder ein Intervall.
\end{block}

\begin{block}{Schritt 2: Hauptsatz}
Dann folgt
\[
v-u=\int_{T^{-1}(u)}^{T^{-1}(v)} T'(x)\,dx.
\]
Damit gilt die Behauptung für Intervall-Indikatoren.
\end{block}

\begin{block}{Schritt 3}
Mit Linearität folgt der Satz für Treppenfunktionen, danach per Approximation für allgemeine Funktionen.
\end{block}
\end{frame}

\begin{frame}{Beispiel in einer Dimension}
\begin{block}{Beispiel}
Berechne
\[
\int_1^2 2x\cos(x^2)\,dx.
\]
Mit
\[
T(x)=x^2,\qquad T'(x)=2x
\]
erhält man
\[
\int_1^2 2x\cos(x^2)\,dx
=
\int_1^4 \cos(y)\,dy
=
\sin(4)-\sin(1).
\]
Die Grenzen ändern sich also sichtbar durch
\[
[1,2]\mapsto [1,4].
\]
\end{block}
\end{frame}

\begin{frame}{Beispiel für Dichtetransformation}
\begin{block}{Beispiel}
Sei \(X\sim U([1,2])\), also mit Dichte
\[
p(x)=\mathbf 1_{[1,2]}(x),
\]
und \(Y=X^2\).
Dann ist die Dichte von \(Y\) auf \([1,4]\):
\[
q(y)=p(\sqrt y)\frac{1}{2\sqrt y}
=
\frac{1}{2\sqrt y}\,\mathbf 1_{[1,4]}(y).
\]
\end{block}

Für messbares \(f\) gilt

\]
Also wird \([1,2]\) auf \([1,4]\) abgebildet.
\end{block}
\end{frame}

\begin{frame}{Stochastische Unabhängigkeit}
\begin{block}{Definition}
Zwei Zufallsvariablen
\[
X\colon \Omega\to E,
\qquad
Y\colon \Omega\to F
\]
hei\ss{}en stochastisch unabhängig, wenn für alle \(A\in\mathcal E\), \(B\in\mathcal F\) gilt:
\[
\mathbb P(X\in A,\;Y\in B)=\mathbb P(X\in A)\,\mathbb P(Y\in B).
\]
\end{block}
\end{frame}

\begin{frame}{Unabhängigkeit von Ereignissen}
\begin{block}{Definition}
Zwei Ereignisse \(A,B\in\mathcal A\) hei\ss{}en stochastisch unabhängig, wenn
\[
\mathbb P(A\cap B)=\mathbb P(A)\cdot\mathbb P(B).
\]
\end{block}

\begin{block}{Zusammenhang mit bedingter Wahrscheinlichkeit}
Ist \(\mathbb P(B)>0\), so gilt:
\[
A \text{ und } B \text{ unabhängig}
\quad\Longleftrightarrow\quad
\mathbb P(A\mid B)=\mathbb P(A).
\]
Das Wissen, dass \(B\) eingetreten ist, ändert die Wahrscheinlichkeit von \(A\) nicht.
\end{block}
\end{frame}

\begin{frame}{Unabhängigkeit von Ereignissen: Beweis}
\begin{block}{``\(\Rightarrow\)''}
Sei \(A\cap B\) unabhängig, d.\,h.\ \(\mathbb P(A\cap B)=\mathbb P(A)\mathbb P(B)\).
Dann:
\[
\mathbb P(A\mid B)=\frac{\mathbb P(A\cap B)}{\mathbb P(B)}=\frac{\mathbb P(A)\mathbb P(B)}{\mathbb P(B)}=\mathbb P(A).
\]
\end{block}

\begin{block}{``\(\Leftarrow\)''}
Sei \(\mathbb P(A\mid B)=\mathbb P(A)\). Dann:
\[
\mathbb P(A\cap B)=\mathbb P(A\mid B)\cdot\mathbb P(B)=\mathbb P(A)\cdot\mathbb P(B).
\]
\end{block}
\end{frame}

\begin{frame}{Klassische Definition als Spezialfall}
\begin{block}{Indikatorfunktion}
Für \(A\in\mathcal A\) gilt
\[
\mathbf{1}_A^{-1}(\{1\})=A,
\qquad
\mathbf{1}_A^{-1}(\{0\})=A^c.
\]
\end{block}

\begin{block}{Herleitung}
Setze \(X=\mathbf{1}_A\), \(Y=\mathbf{1}_B\). Die abstrakte Unabhängigkeit verlangt
\[
\mathbb P(X\in C,\;Y\in D)=\mathbb P(X\in C)\,\mathbb P(Y\in D).
\]
Die linke Seite ist die gemeinsame Verteilung auf dem Rechteck:
\[
\mathbb P(X\in C,\;Y\in D)
=\mathbb P_{(X,Y)}(C\times D)
=\mathbb P\bigl((X,Y)^{-1}(C\times D)\bigr).
\]
\end{block}
\end{frame}

\begin{frame}{Klassische Definition als Spezialfall (Forts.)}
\begin{block}{Urbild des Rechtecks}
Per Definition des Urbilds:
\[
(X,Y)^{-1}(C\times D)=\{\omega\in\Omega : (X(\omega),Y(\omega))\in C\times D\}.
\]
Nun ist \((X(\omega),Y(\omega))\in C\times D\) genau dann, wenn
\[
X(\omega)\in C \quad\text{und}\quad Y(\omega)\in D,
\]
also
\begin{align*}
(X,Y)^{-1}(C\times D)
&=\{\omega: X(\omega)\in C\}\cap\{\omega: Y(\omega)\in D\}\\
&=X^{-1}(C)\cap Y^{-1}(D).
\end{align*}
\end{block}
\end{frame}

\begin{block}{Einsetzen: \(C=\{1\}\), \(D=\{1\}\)}
\[
X^{-1}(\{1\})\cap Y^{-1}(\{1\})=A\cap B,
\]
und damit
\[
\mathbb P(A\cap B)=\mathbb P(A)\cdot\mathbb P(B).
\]
\end{block}
\end{frame}

\begin{frame}{Charakterisierung der Unabhängigkeit}
\begin{block}{Satz}
Für Zufallsvariablen
\[
X\colon \Omega\to E,\qquad Y\colon \Omega\to F
\]
sind äquivalent:
\begin{enumerate}
    \item \(X\) und \(Y\) sind stochastisch unabhängig,
    \item
    \[
    \mathbb P_{(X,Y)}=\mathbb P_X\otimes\mathbb P_Y.
    \]
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{Warum ist das plausibel?}
\begin{block}{Beobachtung}
Für alle \(A\in\mathcal E\), \(B\in\mathcal F\) gilt
\[
(X,Y)\in A\times B
\iff
X\in A \text{ und } Y\in B.
\]
\end{block}

\begin{block}{Daher}
\[
\mathbb P_{(X,Y)}(A\times B)
=
\mathbb P(X\in A,\;Y\in B).
\]
Also kodiert die gemeinsame Verteilung auf Rechtecken genau die gemeinsamen Auftretenswahrscheinlichkeiten.
\end{block}
\end{frame}

\begin{frame}{Beweisidee: \(1\Rightarrow 2\)}
\begin{block}{\(1\Rightarrow 2\)}
Aus der Unabhängigkeit folgt auf jedem Rechteck
\[
\mathbb P_{(X,Y)}(A\times B)
=
\mathbb P_X(A)\mathbb P_Y(B)
=
(\mathbb P_X\otimes\mathbb P_Y)(A\times B).
\]
Da Rechtecke die Produkt-\(\sigma\)-Algebra erzeugen, folgt
\[
\mathbb P_{(X,Y)}=\mathbb P_X\otimes\mathbb P_Y.
\]
\end{block}
\end{frame}

\begin{frame}{Beweisidee: \(2\Rightarrow 1\)}
\begin{block}{\(2\Rightarrow 1\)}
Gilt umgekehrt
\[
\mathbb P_{(X,Y)}=\mathbb P_X\otimes\mathbb P_Y,
\]
so erhält man auf Rechtecken sofort
\[
\mathbb P(X\in A,\;Y\in B)=\mathbb P(X\in A)\mathbb P(Y\in B).
\]
Also sind \(X\) und \(Y\) unabhängig.
\end{block}
\end{frame}
