\begin{frame}
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Satz der totalen Wahrscheinlichkeit}
Für eine Zerlegung  $\Omega = \bigcup_{j=1}^{n} B_j, \text{ mit } B_i \cap B_k = \emptyset \text{ für } i \neq k $
\begin{align*}
& P(A ) = \sum_{j=1}^{n}  P(A \; | \;  B_j) \cdot P(B_j)
\end{align*}
\end{block}

\begin{block}{Beweis}
Da $A = A \cap \Omega = A \cap \bigcup_{j=1}^{n} B_j = \bigcup_{j=1}^{n} (A \cap B_j)$ eine disjunkte Zerlegung ist, folgt:
\begin{align*}
P(A) &= \sum_{j=1}^{n} P(A \cap B_j) = \sum_{j=1}^{n} P(A \; | \; B_j) \cdot P(B_j)
\end{align*}
\end{block}

 \end{frame}



\begin{frame}
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Satz von Bayes}
Für $A,B \in \mathcal{A}$ mit  $P(B) > 0$ gilt
\begin{align*}
& P(A \; | \;  B) = \frac{P(B \; | \; A) \cdot P(A)} {P(B)} \\
\end{align*}
\end{block}

\begin{block}{Beweis}
\begin{align*}
& P(A \; | \;  B) =\frac{P(A \cap B)}{P(B)} = \frac{ \frac{P(A \cap B) \cdot P(A)}{P(A)}}{P(B)}  =  \frac{P(B \; | \; A) \cdot P(A)} {P(B)} 
\end{align*}
\end{block}



 \end{frame}


\begin{frame}
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Bayes'sche Inferenz (Bayesian Inference)}
Die Formel von Bayes beschreibt, wie wir eine Hypothese $H$ nach Beobachtung von Daten $D$ aktualisieren:
\begin{align*}
P(H \mid D) = \frac{P(D \mid H) \cdot P(H)}{P(D)}
\end{align*}
Dabei heißen
\begin{align*}
&P(H) &&\text{Prior}, \\
&P(D \mid H) &&\text{Likelihood}, \\
&P(H \mid D) &&\text{Posterior}.
\end{align*}
\end{block}

 \end{frame}


\begin{frame}[shrink=5]
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Interpretation der Bayes'schen Inferenz}
Wir starten mit einer anfänglichen Einschätzung $P(H)$ über eine Hypothese und passen diese durch neue Daten $D$ an.
Ist $P(D \mid H)$ groß, dann sprechen die Daten für die Hypothese; ist $P(D \mid H)$ klein, dann spricht das gegen sie.
\end{block}

\begin{block}{Prior, Likelihood, Posterior}
\begin{align*}
&\text{Prior } P(H): &&\text{unsere Einschätzung vor den Daten}, \\
&\text{Likelihood } P(D \mid H): &&\text{wie gut erklärt } H \text{ die Daten } D, \\
&\text{Posterior } P(H \mid D): &&\text{unsere Einschätzung nach den Daten}.
\end{align*}
Wichtig: Die Likelihood ist nicht $P(H \mid D)$, sondern $P(D \mid H)$. Sie bewertet also die Daten unter der Annahme, dass die Hypothese gilt.
\end{block}

\begin{block}{Typischer Ablauf}
\begin{enumerate}
\item Formuliere mögliche Hypothesen $H_1,\ldots,H_k$.
\item Lege Prior-Wahrscheinlichkeiten $P(H_i)$ fest.
\item Modelliere die Datenwahrscheinlichkeit $P(D \mid H_i)$.
\item Berechne daraus die Posterior-Wahrscheinlichkeiten $P(H_i \mid D)$.
\item Entscheide z.B. mit der wahrscheinlichsten Hypothese (MAP-Regel).
\end{enumerate}
\end{block}

 \end{frame}


\begin{frame}[shrink=5]
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Beispiel: medizinischer Test}
Sei $H=$ ``Person ist krank'' und $D=$ ``Test ist positiv''.
Angenommen:
\begin{align*}
&P(H)=0.01, \\
&P(D \mid H)=0.99, \\
&P(D \mid H^c)=0.05.
\end{align*}
Dann ist der Prior klein: Nur $1\%$ der Personen sind krank. Die Likelihood $P(D \mid H)=0.99$ sagt: Wenn jemand krank ist, ist der Test fast immer positiv.
\end{block}

\begin{block}{Posterior berechnen}
Mit Bayes folgt
\begin{align*}
P(H \mid D)
&= \frac{P(D \mid H)P(H)}{P(D \mid H)P(H) + P(D \mid H^c)P(H^c)} \\
&= \frac{0.99 \cdot 0.01}{0.99 \cdot 0.01 + 0.05 \cdot 0.99}
\approx 0.167.
\end{align*}
Obwohl der Test gut ist, liegt die Wahrscheinlichkeit nach einem positiven Test also nur bei etwa $16.7\%$.
\end{block}

 \end{frame}


\begin{frame}[shrink=5]
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Bayes bei kontinuierlichen Verteilungen}
Ist der unbekannte Parameter $\Theta$ selbst stetig verteilt, dann arbeiten wir mit Dichten statt mit einzelnen Wahrscheinlichkeiten.
Für Prior-Dichte $p(\theta)$ und Likelihood-Dichte $f(x \mid \theta)$ gilt nach Beobachtung $X=x$:
\begin{align*}
p(\theta \mid x)
= \frac{f(x \mid \theta)\, p(\theta)}{\int_{-\infty}^{\infty} f(x \mid u)\, p(u)\, du}.
\end{align*}
Bei stetigen Verteilungen ist nämlich $P(X=x)=0$; relevant ist daher die Dichte an der beobachteten Stelle $x$.
\end{block}

 \end{frame}


\begin{frame}[shrink=5]
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Beispiel: Raumtemperatur und Thermometer}
Sei $\Theta$ die wahre Raumtemperatur in Grad Celsius. Vor der Messung glauben wir, dass die Temperatur meistens in der Nähe von $20^\circ$C liegt:
\begin{align*}
\Theta \sim \mathcal N(20,2^2).
\end{align*}
Das Thermometer misst nicht perfekt, sondern mit einem Fehler von etwa $1^\circ$C. Also modellieren wir
\begin{align*}
X \mid \Theta=\theta \sim \mathcal N(\theta,1^2).
\end{align*}
Wenn das Thermometer nun $X=24$ misst, dann ergibt sich als Posterior wieder eine Normalverteilung:
\begin{align*}
\Theta \mid X=24 \sim \mathcal N(23.2,0.8).
\end{align*}
\end{block}

\begin{block}{Interpretation}
Der Prior sagt: Vor der Messung erwarten wir ungefähr $20^\circ$C.
Die Likelihood sagt: Ein Messwert von $24^\circ$C ist besonders plausibel, wenn die wahre Temperatur in der Nähe von $24^\circ$C liegt.
Der Posterior kombiniert beides: Nach der Messung halten wir Werte um $23.2^\circ$C für am plausibelsten.
\end{block}

 \end{frame}



\begin{frame}
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Erinnerung Stochastische Unabhängigkeit}
Zwei Ereignisse $A,B$ heißen stochastisch unabhängig, falls
\begin{align*}
P(A \cap B) = P(A) \cdot P(B)
\end{align*}
gilt.  Gleichbedeutend damit ist  $P(A | B) = P(A)$ und $P(B  | A) = P(B)$.
\end{block}

 \end{frame}

(Typische Wörter wären zum Beispiel "reich",  "casino", "Vergrösserung" ...)
\end{block}

\begin{frame}
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Naiver Bayes'scher Spam Filter}

Aus einer Datenbank kann man das Vorkommen dieser Wörter in Spam und Ham Mails zählen und damit empirisch die Wahrscheinlichkeiten $P(A_i | S)$ und $P(A_i | H) $ des Vorkommens dieser Wörter in Spam und Ham Mails ermitteln.  Wir gehen davon aus, dass es sich bei der Mail  prinzipiell mit  Wahrscheinlichkeit $P(E= S) = P(E= H)= \frac{1}{2}$  um eine erwünschte  Mail $H$ oder eine unerwünschte Mail $S$  handeln kann. 
\end{block}

 \end{frame}



\begin{frame}
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Naiver Bayes'scher Spam Filter}
 Wir machen zudem die (naive) Annahme, dass das Vorkommen der Wörter  stochastisch unabhängig ist, also 
\begin{align*}
P(A_1 \cap \cdots \cap A_n | S) = P(A_1 | S) \cdot P(A_2 | S) \cdots P(A_n | S) \\
P(A_1 \cap \cdots \cap A_n | H) = P(A_1 | H) \cdot P(A_2 | H) \cdots P(A_n | H)
\end{align*}
gilt.
\end{block}

 \end{frame}


\begin{frame}
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Naiver Bayes'scher Spam Filter}
Mit der Formel von Bayes, der totalen Wahrscheinlichkeit  und kürzen, da $P(S) = P(H)$ können wir somit berechnen
\begin{align*}
& P(E=S |  A_1 \cap \cdots \cap A_n) = \frac{P(A_1 \cap \cdots \cap A_n | S) \cdot P(S)}{P(A_1 \cap \cdots \cap A_n)} \\
&=  \frac{P(A_1 | S) \cdots P(A_n | S) }{P(A_1 \cap \cdots \cap A_n | H) + P(A_1 \cap \cdots \cap A_n | S)} \\
&=  \frac{P(A_1 | S) \cdots P(A_n | S) }{P(A_1 | H) \cdots P(A_n | H)  + P(A_1 | S) \cdots P(A_n | S) } \\
\end{align*}
Bemerkung: $P(E=H |  A_1 \cap \cdots \cap A_n) = 1- P(E=S |  A_1 \cap \cdots \cap A_n) $
\end{block}



 \end{frame}


\begin{frame}[shrink=5]
    \frametitle{Diskrete Wahrscheinlichkeitsverteilung}
\framesubtitle{}

\begin{block}{Spamfilter als Bayes'sche Inferenz}
Der naive Bayes-Filter ist ein klassisches Beispiel für Bayes'sche Inferenz:
\begin{align*}
&\text{Hypothesen: } H=\text{Ham},\ S=\text{Spam}, \\
&\text{Daten: } D = A_1 \cap \cdots \cap A_n.
\end{align*}
Aus Prior und Likelihoods berechnen wir die Posterior-Wahrscheinlichkeiten
\begin{align*}
P(S \mid D), \qquad P(H \mid D).
\end{align*}
\end{block}

\begin{block}{Einordnung}
\begin{itemize}
\item Prior: $P(S), P(H)$ beschreibt, wie wahrscheinlich Spam oder Ham vor den Wörtern ist.
\item Likelihood: $P(D \mid S), P(D \mid H)$ misst, wie gut die beobachteten Wörter zu Spam oder Ham passen.
\item Posterior: $P(S \mid D), P(H \mid D)$ kombiniert beides nach der Bayes-Formel.
\item Entscheidung: Klassifiziere als Spam, falls $P(S \mid D) > P(H \mid D)$.
\end{itemize}
\end{block}

 \end{frame}





 \begin{frame}
    \frametitle{Erwartungswert}
\framesubtitle{}
\begin{block}{Eigenschaften}
Sind $X,Y : \Omega \to \mathbb{R}^n$   reelle, integrierbare  Zufallsvariablen und $a,b \in \mathbb{R}$ konstant, so gilt:
\begin{align*}
& \mathbb{E}(a \cdot X + b \cdot Y) = a \cdot \mathbb{E}(X) + b \cdot \mathbb{E}(Y) \\
& X(x) \leq Y(x) \;  \forall x \in \Omega \Rightarrow \mathbb{E}(X) \leq \mathbb{E}(Y) \\
& X ,Y \text{ stoch. unabhängig} \Rightarrow   \mathbb{E}(X \cdot Y) =  \mathbb{E}(X) \cdot  \mathbb{E}(Y) \\
& \mathbb{E} (1_A) = P (A)
\end{align*}
\end{block}
 \end{frame}

 \begin{frame}
    \frametitle{Erwartungswert}
\framesubtitle{}
\begin{block}{Varianz}
Für eine reelle Zufallsvariable ist die Varianz definiert durch
$$ \mathbb{V} (X) :=  \mathbb{E}( (X - \mathbb{E}(X))^2) \; .$$
\end{block}
\begin{block}{Verschiebungssatz}

\begin{align*}
 \mathbb{V}(X) & = \mathbb{E}(X^2 - 2X \mathbb{E}(X) + \mathbb{E}(X)^2) = \mathbb{E}(X^2) - 2 \mathbb{E}(X)^2 +  \mathbb{E}(X)^2 \\
& =  \mathbb{E}(X^2) -  \mathbb{E}(X)^2 \\
\end{align*}
\end{block}
 \end{frame}

 \begin{frame}
    \frametitle{Erwartungswert}
\framesubtitle{}
\begin{block}{Kovarianz}
Für  reelle Zufallsvariable $X,Y$ ist die Kovarianz definiert durch
$$ \mathcal{C} (X,Y) :=  \mathbb{E}( (X - \mathbb{E}(X)) (Y - \mathbb{E}(Y))) \; .$$
\end{block}

\begin{block}{Kovarianz}
    Per Definition ist
    $$ \mathcal{C} (X,X) :=  \mathbb{V}(X).$$
    \end{block}
 \end{frame}

\begin{frame}
    \frametitle{Erwartungswert}
\framesubtitle{}
\begin{block}{Markov Ungleichung}
Sei $Y : \Omega \to \mathbb{R}$  eine  reelle, integrierbare  Zufallsvariable und $f : [0, \infty) \to [0, \infty)$ monoton wachsend.
Dann gilt für alle $\epsilon > 0$ mit $f(\epsilon) > 0$
\begin{align*}
P (|Y |  \geq \epsilon) \leq \frac{\mathbb{E} (f \circ |Y|)}{f(\epsilon)}
\end{align*}
\end{block}
\begin{block}{Beweis}
Da $f(\epsilon) 1_{\{ |Y| \geq  \epsilon \} } \leq f \circ |Y|$ folgt
\begin{align*}
f(\epsilon) P(|Y| \geq \epsilon) = & f(\epsilon) \mathbb{E}(1_{\{ |Y| \geq  \epsilon \} }) = \mathbb{E}( f(\epsilon) 1_{\{ |Y| \geq  \epsilon \} }) \\
\leq & \mathbb{E}( f \circ |Y|)
\end{align*}
\end{block}
 \end{frame}






\begin{frame}
    \frametitle{Erwartungswert}
\framesubtitle{}
\begin{block}{Tschebyscheff-Ungleichung}
Für eine reelle, integrierbare und quadratintegrierbare  Zufallsvariable $Y : \Omega \to \mathbb{R}$  gilt:
\begin{align*}
P (|Y  - \mathbb{E} (Y)|  \geq \epsilon) \leq \frac{\mathbb{V} (Y)}{ \epsilon^2} 
\end{align*}
\end{block}
\begin{block}{Beweis}
Folgt direkt aus der Markov-Ungleichung mit $Y' = Y -\mathbb{E}(Y)$ und $f(x) = x^2$
\end{block}
 \end{frame}



\begin{frame}
    \frametitle{Erwartungswert}
\framesubtitle{}
\begin{block}{Schwaches Gesetz der großen Zahlen}
Seien $X_i : \Omega \to \mathbb{R}$ unabhängige, reelle Zufallsvariablen (uiv, iid(englisch)) mit $\mathbb{E}(X_i) = \mu < \infty$ und $\mathbb{V}(X_i) = \sigma < \infty$, dann gilt
\begin{align*}
P \bigl  ( \bigl | \frac{1}{n} \sum_{i=1}^{n} X_i - \mu \bigr |  \geq \epsilon \bigr) \leq \frac{\sigma}{ n \cdot \epsilon^2} \; \; \underset{n \to \infty}{\longrightarrow} 0
\end{align*}
(stochastische Konvergenz). 
\end{block}
 \end{frame}


\begin{frame}
    \frametitle{Erwartungswert}
\framesubtitle{}
\begin{block}{Beweis}
Mit $Y_n =  \frac{1}{n} \sum_{i=1}^{n}  X_i - \mu$ ist $\mathbb{E}(Y_n) =  \frac{1}{n} \sum_{i=1}^{n} \mathbb{E}( X_i - \mu) = 0$ und 
$\mathbb{V}(Y_n) =  \frac{1}{n^2} \sum_{i=1}^{n} \mathbb{V}( X_i ) = \frac{\sigma}{n}$. Aus der Tschebyscheff-Ungleichung folgt die Behauptung.
\end{block}
 \end{frame}
